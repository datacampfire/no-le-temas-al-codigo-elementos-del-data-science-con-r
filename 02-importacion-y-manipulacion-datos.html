group_by + summarize Resumir Columnas por Grupos

group_by + summarize Ejemplo

group_by + summarise Código

pollution %>%
  group_by(city) %>% 
  summarise(
    promedio = median(amount),
    suma = sum(amount),
    n = n()
    )
## # A tibble: 3 x 4

## 3 New York     18.5    37     2
